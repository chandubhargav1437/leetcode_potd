#include <vector>
#include <string>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    long long minimumCost(string source, string target, vector<char>& original, vector<char>& changed, vector<int>& cost) {
        // Use a large constant for infinity.
        // LLONG_MAX / 2 prevents overflow when adding two costs together.
        const long long INF = LLONG_MAX / 2;
        
        // 26x26 matrix to store minimum conversion costs.
        // dist[i][j] represents the cost to change char (i+'a') to char (j+'a').
        vector<vector<long long>> dist(26, vector<long long>(26, INF));

        // The cost to change a character to itself is always 0.
        for (int i = 0; i < 26; ++i) {
            dist[i][i] = 0;
        }

        // 1. Build the initial graph from the input arrays.
        for (size_t i = 0; i < original.size(); ++i) {
            int u = original[i] - 'a';
            int v = changed[i] - 'a';
            // We take the minimum because the input might contain multiple 
            // costs for the same transformation.
            dist[u][v] = min(dist[u][v], (long long)cost[i]);
        }

        // 2. Apply Floyd-Warshall Algorithm to find all-pairs shortest paths.
        // k is the intermediate node.
        for (int k = 0; k < 26; ++k) {
            for (int i = 0; i < 26; ++i) {
                for (int j = 0; j < 26; ++j) {
                    // If passing through k is cheaper, update the path.
                    if (dist[i][k] != INF && dist[k][j] != INF) {
                        dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
                    }
                }
            }
        }

        // 3. Calculate total cost for converting source to target.
        long long totalCost = 0;
        int n = source.length();

        for (int i = 0; i < n; ++i) {
            int u = source[i] - 'a';
            int v = target[i] - 'a';

            // If the characters are different, we check the cost.
            if (u != v) {
                if (dist[u][v] == INF) {
                    return -1; // Conversion is impossible
                }
                totalCost += dist[u][v];
            }
        }

        return totalCost;
    }
};
