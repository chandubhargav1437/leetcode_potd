/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
    private:
    long long totalSum = 0;
    long long maxxProduct = 0;

    // Helper function to calculate subtree sums
    // It returns the sum of the subtree rooted at 'node'
    long long calculateSubtreeSum(TreeNode* node) {
        if (node == nullptr) return 0;

        // Recursively get sum of left and right subtrees
        long long leftSum = calculateSubtreeSum(node->left);
        long long rightSum = calculateSubtreeSum(node->right);

        // Current subtree sum = node value + left child sum + right child sum
        long long currentSubtreeSum = node->val + leftSum + rightSum;

        // CRITICAL OPTIMIZATION LOGIC:
        // If totalSum is already calculated (i.e., we are in the second pass),
        // we check if cutting the edge above this node produces a max product.
        if (totalSum != 0) {
            long long remainingSum = totalSum - currentSubtreeSum;
            // Update global maximum product
            maxxProduct = max(maxxProduct, currentSubtreeSum * remainingSum);
        }

        return currentSubtreeSum;
    }
public:
    int maxProduct(TreeNode* root) {
        // Reset global variables for safety
        totalSum = 0;
        maxxProduct = 0;

        // PASS 1: Calculate the Total Sum of the entire tree
        // The maxProduct logic inside won't trigger meaningfully yet because totalSum is 0
        totalSum = calculateSubtreeSum(root);

        // PASS 2: Re-calculate sums. 
        // Now that totalSum is known, the function will update maxProduct at every node.
        calculateSubtreeSum(root);

        // Return result modulo 10^9 + 7
        return (int)(maxxProduct % 1000000007);
    }
};
