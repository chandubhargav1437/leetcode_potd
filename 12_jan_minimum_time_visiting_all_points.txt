class Solution {
public:
    int minTimeToVisitAllPoints(vector<vector<int>>& points) {
        int totalTime = 0;
        int n = points.size();

        // Iterate through the sequence of points
        for (int i = 0; i < n - 1; ++i) {
            // Current point coordinates
            int x1 = points[i][0];
            int y1 = points[i][1];
            
            // Next point coordinates
            int x2 = points[i+1][0];
            int y2 = points[i+1][1];

            // Calculate absolute differences
            int dx = abs(x2 - x1);
            int dy = abs(y2 - y1);

            // The time to travel between two points is the maximum 
            // of the differences in x and y coordinates (Chebyshev distance).
            totalTime += max(dx, dy);
        }

        return totalTime;
    }
};
