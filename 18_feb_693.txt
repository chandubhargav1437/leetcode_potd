class Solution {
public:
    bool hasAlternatingBits(int n) {
        long long ress=1;
        long long temp=2;
        while(temp<n){
        ress++;
            temp=temp*2;
        }
      string num=bitset<31>(n).to_string().substr(31-ress);
    for(int i=0;i<num.size()-1;i++){

        if(num[i]==num[i+1]){
            return false;
        }
    }
      return true;
    }
};
