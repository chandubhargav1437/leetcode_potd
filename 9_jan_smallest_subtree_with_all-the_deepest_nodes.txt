/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    // Helper function type: returns a pair {depth, node}
    // depth: the maximum depth found in this subtree
    // node: the root of the smallest subtree containing all deepest nodes
    pair<int, TreeNode*> dfs(TreeNode* root) {
        // Base case: If node is null, depth is 0 and no subtree exists.
        if (root == nullptr) {
            return {0, nullptr};
        }

        // Recursive step: Post-order traversal (Process children first)
        pair<int, TreeNode*> leftResult = dfs(root->left);
        pair<int, TreeNode*> rightResult = dfs(root->right);

        int leftDepth = leftResult.first;
        int rightDepth = rightResult.first;

        // Logic to determine which node to return
        if (leftDepth == rightDepth) {
            // If both sides are equally deep, the current 'root' is the common ancestor.
            // We add 1 to the depth to account for the current level.
            return {leftDepth + 1, root};
        } else if (leftDepth > rightDepth) {
            // If left is deeper, the answer is somewhere in the left subtree.
            // We preserve the 'node' returned by the left call but update depth.
            return {leftDepth + 1, leftResult.second};
        } else {
            // If right is deeper, the answer is in the right subtree.
            return {rightDepth + 1, rightResult.second};
        }
    }

    TreeNode* subtreeWithAllDeepest(TreeNode* root) {
        // We only care about the node part of the result pair
        return dfs(root).second;
    }
};
