#include <vector>
#include <string>

using namespace std;

class Solution {
public:
    vector<string> readBinaryWatch(int turnedOn) {
        vector<string> valid_times;
        
        // A watch can only show hours from 0 to 11
        for (int h = 0; h < 12; h++) {
            // A watch can only show minutes from 0 to 59
            for (int m = 0; m < 60; m++) {
                // Count the number of set bits (1s) in the hour and minute
                // __builtin_popcount is a built-in function that counts set bits
                if (__builtin_popcount(h) + __builtin_popcount(m) == turnedOn) {
                    
                    // Format the time string according to the rules
                    string time = to_string(h) + (m < 10 ? ":0" : ":") + to_string(m);
                    valid_times.push_back(time);
                }
            }
        }
        
        return valid_times;
    }
};
