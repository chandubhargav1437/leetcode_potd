class Solution {
public:
    int concatenatedBinary(int n) {
        long long ans = 0;
        int MOD = 1e9 + 7;
        int length = 0; // Tracks the binary length of the current number i
        
        for (int i = 1; i <= n; i++) {
            // If i is a power of 2, its binary length increases by 1
            if ((i & (i - 1)) == 0) {
                length++;
            }
            
            // Shift the current answer left by 'length' bits, add 'i', and take modulo
            ans = ((ans << length) | i) % MOD;
        }
        
        return ans;
    }
};
