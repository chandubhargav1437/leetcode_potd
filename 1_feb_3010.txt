class Solution {
public:
    int minimumCost(vector<int>& nums) {
        // 1. The first element is always the head of the first subarray.
        int totalCost = nums[0];
        
        // 2. We need to find the two smallest elements in the remaining array (index 1 to end).
        // Initialize min1 and min2 to the maximum possible integer value.
        int min1 = 1000; // Based on typical constraints, or use INT_MAX
        int min2 = 1000; 

        // Iterate through the rest of the array
        for (int i = 1; i < nums.size(); i++) {
            if (nums[i] < min1) {
                // If we find a new smallest, push the old smallest to second place
                min2 = min1;
                min1 = nums[i];
            } else if (nums[i] < min2) {
                // If it's not smaller than min1 but smaller than min2, update min2
                min2 = nums[i];
            }
        }
        
        // 3. Return the sum of the first element and the two smallest found.
        return totalCost + min1 + min2;
    }
};
