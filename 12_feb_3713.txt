#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

class Solution {
public:
    int longestBalanced(string s) {
        int n = s.length();
        int maxLen = 0;

        for (int i = 0; i < n; ++i) {
            // Optimization: If the remaining substring length is less than 
            // the maxLen we've already found, we can't find a better one.
            if (n - i <= maxLen) break;

            vector<int> freq(26, 0);
            int distinctCount = 0;
            int maxFreq = 0;

            for (int j = i; j < n; ++j) {
                int charIdx = s[j] - 'a';
                
                // If this is the first time we see this character in the current window
                if (freq[charIdx] == 0) {
                    distinctCount++;
                }
                
                freq[charIdx]++;
                maxFreq = max(maxFreq, freq[charIdx]);

                // Current length of the substring window
                int currentLen = j - i + 1;

                // Check if balanced: 
                // If all distinct characters appear 'maxFreq' times, 
                // the total length must be distinctCount * maxFreq.
                if (maxFreq * distinctCount == currentLen) {
                    maxLen = max(maxLen, currentLen);
                }
            }
        }
        return maxLen;
    }
};
