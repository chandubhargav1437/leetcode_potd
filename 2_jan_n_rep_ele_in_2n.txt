class Solution {
public:
    int repeatedNTimes(vector<int>& nums) {
        int n=nums.size();
        // int i=0;
        for(int i=0;i<n;++i){
            // if(i<=n-1){

            if(i+1<n&&nums[i]==nums[i+1]){
                return nums[i];
            }
                if(i+2<n&&nums[i]==nums[i+2]){
                return nums[i];
            }
        }
        return nums[0];
    }
};
